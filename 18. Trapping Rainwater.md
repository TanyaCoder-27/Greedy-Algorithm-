## Trapping Rainwater

```
import java.util.*;
class Main {
    public static void main(String[] args) {
       Scanner sc=new Scanner(System.in);
       System.out.println("Enter no of buildings:");
       int n=sc.nextInt();
       int a[]=new int[n];
        System.out.println("Enter heights of buildings:");
       for(int i=0;i<n;i++){
           a[i]=sc.nextInt();
       }
       
       System.out.println(trapWater(a,n));
    }
    static int trapWater(int a[],int n)
    {
        int lmax=a[0],rmax=a[n-1],i=1,j=n-2,ans=0;
        while(i<=j)
        {
            if(rmax<=lmax)
            {
                ans+=Math.max(0,rmax-a[i]);
                if(a[i]>rmax)
                rmax=a[i];
                j--;
            }
            else
            {
                ans+=Math.max(0,lmax-a[i]);
                if(a[i]>lmax)
                lmax=a[i];
                i++;
            }
        }
        return ans;
    }
}
```
