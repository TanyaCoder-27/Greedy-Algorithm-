## 8.Product of array except itself without division operator.

```
import java.util.*;
class Main {
    public static void main(String[] args) {
        Scanner sc=new Scanner(System.in);
        System.out.println("enter n");
        int n=sc.nextInt();
        int a[]=new int[n];
        for(int i=0;i<n;i++)
        a[i]=sc.nextInt();
       int res[]= eleProd(a,n);
        System.out.println("product of elements except itself are ");    
       for(int i=0;i<n;i++)
        System.out.println(res[i]+" ");    
    }
    static int[] eleProd(int a[],int n)
    {
        int m=1;
        int res[]=new int[n];
        int l1[]=new int[n];
        int l2[]=new int[n];
        l1[0]=1;
        for(int i=1;i<n;i++)
        {
            l1[i]=l1[i-1]*a[i-1];
        }
        l2[n-1]=1;
        for(int i=n-2;i>=0;i--)
        {
            l2[i]=l2[i+1]*a[i+1];
        }
        for(int i=0;i<n;i++)
        res[i]=l1[i]*l2[i];
        return res;
    }
}
```
