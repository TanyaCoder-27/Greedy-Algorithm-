## 9.Distributing candies.
```
import java.util.*;
class Main {
    public static void main(String[] args) {
        Scanner sc=new Scanner(System.in);
        int n=sc.nextInt();
        int a[]=new int[n];
        for(int i=0;i<n;i++)
        {
            a[i]=sc.nextInt();
        }
        int l1[]=new int [n];
        int l2[]=new int[n];
        for(int i=0;i<n;i++)
        {
            l1[i]=1;
            l2[i]=1;
        }
        for(int i=1;i<n;i++)
        {
            if(a[i]>a[i-1])
            l1[i]=l1[i-1]+1;
        }
        for(int i=n-2;i>=0;i--)
        {
            if(a[i]>a[i+1])
            l2[i]=l2[i+1]+1;
        }
        int s=0;
        for(int i=0;i<n;i++)
        {
            if(l1[i]>l2[i])
            s+=l1[i];
            else
            s+=l2[i];
        }
        System.out.println(s);
    }
}
```
